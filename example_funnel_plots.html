sample_pcts <- sample(sample_data$pct, 6)

walk(
  sample_pcts,
  ~print(
    make_funnel_phe(
      sample_data,
      numerator = numerator,
      denominator = denominator,
      highlighted = pct == .x
    ) +
      ggtitle(.x)
  )
)
